    <title>Spelling Game</title>

    <style>

        body { font-family: Arial, sans-serif; text-align: center; }

        #word-display { font-size: 24px; font-weight: bold; margin: 20px; }

        #feedback { font-size: 20px; margin-top: 10px; }

        input { font-size: 18px; padding: 5px; }

        button { font-size: 18px; padding: 5px 10px; cursor: pointer; }

    </style>

</head>

<body>

    <h1>Spelling Practice Game</h1>

    <div id="word-display">Press Start to See the Word</div>

    <input type="text" id="user-input" placeholder="Type the word here" disabled>

    <button onclick="checkSpelling()">Submit</button>

    <button onclick="startGame()">Start</button>

    <div id="feedback"></div>


    <script>

        const words = ["apple", "banana", "cherry", "grape", "orange", "pineapple", "strawberry"];

        let currentWord = "";


        function startGame() {

            document.getElementById("feedback").innerText = "";

            document.getElementById("user-input").value = "";

            document.getElementById("user-input").disabled = true;

            currentWord = words[Math.floor(Math.random() * words.length)];

            document.getElementById("word-display").innerText = currentWord;

            setTimeout(() => {

                document.getElementById("word-display").innerText = "Now type the word!";

                document.getElementById("user-input").disabled = false;

                document.getElementById("user-input").focus();

            }, 2000); // Show word for 2 seconds

        }


        function checkSpelling() {

            let userInput = document.getElementById("user-input").value.trim().toLowerCase();

            if (userInput === currentWord) {

                document.getElementById("feedback").innerText = "✅ Correct!";

            } else {

                document.getElementById("feedback").innerText = "❌ Try Again! The word was: " + currentWord;

            }

        }

    </script>

</body>
